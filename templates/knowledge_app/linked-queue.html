🔗 链队列 (Linked Queue)

基于链表实现的动态队列，无容量限制，灵活高效

📋

知识点概述

核心概念：链队列是用链表实现的队列，通过动态分配节点来存储数据，没有固定容量限制，遵循"先进先出(FIFO)"原则。

🎯 队头指针(Front)

指向队列第一个节点，用于出队操作

🎯 队尾指针(Rear)

指向队列最后一个节点，用于入队操作

🎯 节点(Node)

包含数据域和指针域的基本单元

🎯 动态分配

根据需要动态创建和销毁节点

进阶级前置知识：链表基础、指针操作、动态内存管理

🔍

概念详解

🚂 生活类比：火车车厢连接

想象一列火车：

  • 车头 = 队头指针（指向第一节车厢）
  • 车尾 = 队尾指针（指向最后一节车厢）
  • 车厢 = 节点（载客 + 连接下一节车厢）
  • 加车厢 = 入队（在尾部连接新车厢）
  • 摘车厢 = 出队（从头部摘掉车厢）

💡 关键：火车可以无限加车厢，不受站台长度限制！

🆚 链队列 vs 顺序队列

📚 顺序队列

  • ✅ 访问速度快，无指针开销
  • ✅ 实现简单，内存连续
  • ❌ 容量固定，可能假溢出
  • ❌ 浪费空间，扩容困难

🔗 链队列

  • ✅ 容量动态，无假溢出
  • ✅ 节省空间，按需分配
  • ❌ 额外指针开销
  • ❌ 内存不连续，缓存不友好
⚠️ 初学者易混淆点：
  • 忘记更新front/rear指针：删除最后一个元素时
  • 内存泄漏：出队时忘记释放节点内存
  • 空队列判断：front为NULL时队列为空
  • 指针顺序：要先保存next再删除当前节点

代码实现

#include <iostream>
using namespace std;

// 链队列节点结构
struct Node {
    int data;
    Node* next;

    Node(int value) : data(value), next(nullptr) {}
};

class LinkedQueue {
private:
    Node* front;  // 队头指针
    Node* rear;   // 队尾指针
    int size;     // 队列大小

public:
    LinkedQueue() {
        front = rear = nullptr;
        size = 0;
    }

    ~LinkedQueue() {
        clear();
    }

    // 入队操作
    void enqueue(int data) {
        Node* newNode = new Node(data);

        if (rear == nullptr) {
            // 队列为空，front和rear都指向新节点
            front = rear = newNode;
        } else {
            // 队列不为空，新节点连接到rear后面
            rear->next = newNode;
            rear = newNode;
        }
        size++;
        cout << "入队成功: " << data << endl;
    }

    // 出队操作
    int dequeue() {
        if (front == nullptr) {
            cout << "队列为空!" << endl;
            return -1;
        }

        Node* temp = front;
        int data = front->data;
        front = front->next;

        // 如果队列变空，rear也要置为nullptr
        if (front == nullptr) {
            rear = nullptr;
        }

        delete temp;
        size--;
        cout << "出队成功: " << data << endl;
        return data;
    }

    // 查看队头元素
    int peek() {
        if (front == nullptr) {
            cout << "队列为空!" << endl;
            return -1;
        }
        return front->data;
    }

    // 判断队列是否为空
    bool isEmpty() {
        return front == nullptr;
    }

    // 获取队列大小
    int getSize() {
        return size;
    }

    // 清空队列
    void clear() {
        while (!isEmpty()) {
            dequeue();
        }
    }

    // 显示队列
    void display() {
        if (isEmpty()) {
            cout << "队列为空" << endl;
            return;
        }

        cout << "队列内容: ";
        Node* current = front;
        while (current != nullptr) {
            cout << current->data << " ";
            current = current->next;
        }
        cout << endl;
    }
};
// 链队列节点类
class Node {
    int data;
    Node next;

    public Node(int data) {
        this.data = data;
        this.next = null;
    }
}

public class LinkedQueue {
    private Node front;  // 队头指针
    private Node rear;   // 队尾指针
    private int size;    // 队列大小

    public LinkedQueue() {
        front = rear = null;
        size = 0;
    }

    // 入队操作
    public void enqueue(int data) {
        Node newNode = new Node(data);

        if (rear == null) {
            // 队列为空，front和rear都指向新节点
            front = rear = newNode;
        } else {
            // 队列不为空，新节点连接到rear后面
            rear.next = newNode;
            rear = newNode;
        }
        size++;
        System.out.println("入队成功: " + data);
    }

    // 出队操作
    public int dequeue() {
        if (front == null) {
            System.out.println("队列为空!");
            return -1;
        }

        int data = front.data;
        front = front.next;

        // 如果队列变空，rear也要置为null
        if (front == null) {
            rear = null;
        }

        size--;
        System.out.println("出队成功: " + data);
        return data;
    }

    // 查看队头元素
    public int peek() {
        if (front == null) {
            System.out.println("队列为空!");
            return -1;
        }
        return front.data;
    }

    // 判断队列是否为空
    public boolean isEmpty() {
        return front == null;
    }

    // 获取队列大小
    public int getSize() {
        return size;
    }

    // 显示队列
    public void display() {
        if (isEmpty()) {
            System.out.println("队列为空");
            return;
        }

        System.out.print("队列内容: ");
        Node current = front;
        while (current != null) {
            System.out.print(current.data + " ");
            current = current.next;
        }
        System.out.println();
    }
}
class Node:
    """链队列节点类"""
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedQueue:
    """链队列类"""
    def __init__(self):
        self.front = None  # 队头指针
        self.rear = None   # 队尾指针
        self.size = 0      # 队列大小

    def enqueue(self, data):
        """入队操作"""
        new_node = Node(data)

        if self.rear is None:
            # 队列为空，front和rear都指向新节点
            self.front = self.rear = new_node
        else:
            # 队列不为空，新节点连接到rear后面
            self.rear.next = new_node
            self.rear = new_node

        self.size += 1
        print(f"入队成功: {data}")

    def dequeue(self):
        """出队操作"""
        if self.front is None:
            print("队列为空!")
            return None

        data = self.front.data
        self.front = self.front.next

        # 如果队列变空，rear也要置为None
        if self.front is None:
            self.rear = None

        self.size -= 1
        print(f"出队成功: {data}")
        return data

    def peek(self):
        """查看队头元素"""
        if self.front is None:
            print("队列为空!")
            return None
        return self.front.data

    def is_empty(self):
        """判断队列是否为空"""
        return self.front is None

    def get_size(self):
        """获取队列大小"""
        return self.size

    def display(self):
        """显示队列"""
        if self.is_empty():
            print("队列为空")
            return

        current = self.front
        result = []
        while current:
            result.append(str(current.data))
            current = current.next

        print("队列内容:", " -> ".join(result))

# 使用示例
if __name__ == "__main__":
    queue = LinkedQueue()

    # 测试入队
    queue.enqueue(10)
    queue.enqueue(20)
    queue.enqueue(30)
    queue.display()

    # 测试出队
    queue.dequeue()
    queue.display()

    print(f"队头元素: {queue.peek()}")
    print(f"队列大小: {queue.get_size()}")

💡 选择指南

何时选择链队列：
  • 队列大小无法预估或变化很大
  • 需要节省内存，避免空间浪费
  • 对插入删除性能要求高
  • 不关心内存访问的局部性

学习检验

🤔 问题1：在空队列中入队一个元素后，front和rear指针指向哪里？

答案：front和rear都指向同一个节点（新创建的节点）
解析：
  • 空队列时：front = rear = NULL
  • 第一次入队：front = rear = newNode
  • 这确保了队列的正确状态维护

🤔 问题2：出队最后一个元素时需要注意什么？

答案：需要将rear指针也置为NULL
解析：当front移动到NULL时，说明队列已空，此时rear也必须置为NULL，否则会导致状态不一致。正确代码： 
if (front == NULL) rear = NULL;

🤔 问题3：链队列相比顺序队列的主要优势是什么？

答案：动态容量，无假溢出，按需分配内存
详细对比：
  • 容量：链队列无固定上限，顺序队列容量固定
  • 内存：链队列按需分配，顺序队列可能浪费空间
  • 溢出：链队列无假溢出，顺序队列需要循环处理

🤔 问题4：链队列的时间和空间复杂度分析？

时间复杂度：
  • 入队：O(1) - 直接在rear后添加
  • 出队：O(1) - 直接删除front节点
  • 查看队头：O(1) - 直接访问front
空间复杂度：O(n) - n个节点 + 每个节点额外的指针开销

🤔 问题5：如何避免链队列的内存泄漏？

答案：出队时及时释放节点内存，析构函数清空队列
关键点：
  • 出队时：先保存要删除的节点指针，更新front后再delete
  • 析构函数：循环出队直到队列为空
  • 异常安全：使用RAII原则管理资源
